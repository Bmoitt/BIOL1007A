data("starwars")

SWSpecies <- starwars %>%
  group_by(species) %>%
  summarize(TotalN=n()) %>%
  filter(TotalN >1) %>%
  select(species)

starwars1 <-starwars %>%
  filter(species %in% SWSpecies$species)
